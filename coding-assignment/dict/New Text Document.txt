import math


def addition(num1, num2):
    return num1 + num2


def subtraction(num1, num2):
    return num1 - num2


def multiplication(num1, num2):
    return num1*num2


def division(num1, num2):
    try:
        return num1 / num2
    except ZeroDivisionError:
        print("Cant divide by 0")


def square_root(num1):
    try:
        return math.sqrt(num1)
    except ValueError:
        print("no square root of negative numbers")

    q_num = 0
    correct_guesses = 0
    for x in questions:
        print(x)
        for y in options[q_num]:
            print(y)
        guess = input("Choose the correct option: ").upper()
        correct_guesses += check_answers(questions.get(x), guess)
        q_num += 1
    display_score(correct_guesses)

def check_answers(answer, guess):
    if answer == guess:
        print("CORRECT")
        return 1
    else:
        print("WRONG")
        return 0

    final_score = num1/len(questions)*100
    print("Final Score is "+str(final_score)+"%")

        return True

        return False
